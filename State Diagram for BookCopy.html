from dataclasses import dataclass
from typing import Optional

@dataclass
class Book:
    isbn: str
    title: str
    author: str

@dataclass
class BookCopy:
    copy_id: str
    isbn: str
    status: str = "Available"  # Available, Issued

@dataclass
class Member:
    member_id: str
    name: str
    max_loans: int = 3

@dataclass
class Loan:
    loan_id: str
    copy_id: str
    member_id: str
    issue_date: str
    due_date: str
    return_date: Optional[str] = None

    def is_active(self) -> bool:
        return self.return_date is None

    def mark_returned(self, date: str) -> None:
        self.return_date = date

from typing import Dict, List, Optional
from models import Book, BookCopy, Member, Loan

class Repository:
    def __init__(self) -> None:
        self.books: Dict[str, Book] = {}
        self.copies: Dict[str, BookCopy] = {}
        self.members: Dict[str, Member] = {}
        self.loans: Dict[str, Loan] = {}

    # --- Book / Copy ---
    def add_book(self, book: Book) -> None:
        self.books[book.isbn] = book

    def add_copy(self, copy: BookCopy) -> None:
        self.copies[copy.copy_id] = copy

    def find_available_copy(self, isbn: str) -> Optional[BookCopy]:
        for c in self.copies.values():
            if c.isbn == isbn and c.status == "Available":
                return c
        return None

    def set_copy_status(self, copy_id: str, status: str) -> None:
        self.copies[copy_id].status = status

    # --- Member ---
    def add_member(self, member: Member) -> None:
        self.members[member.member_id] = member

    def find_member(self, member_id: str) -> Optional[Member]:
        return self.members.get(member_id)

    # --- Loans ---
    def create_loan(self, loan: Loan) -> None:
        self.loans[loan.loan_id] = loan

    def count_active_loans(self, member_id: str) -> int:
        return sum(1 for l in self.loans.values() if l.member_id == member_id and l.is_active())

    def find_active_loan(self, copy_id: str, member_id: str) -> Optional[Loan]:
        for l in self.loans.values():
            if l.copy_id == copy_id and l.member_id == member_id and l.is_active():
                return l
        return None

from datetime import date, timedelta
from models import Loan
from repository import Repository

class LibraryService:
    def __init__(self, repo: Repository) -> None:
        self.repo = repo

    def issue_book(self, isbn: str, member_id: str) -> str:
        member = self.repo.find_member(member_id)
        if not member:
            return "ERROR: Member not found."

        active = self.repo.count_active_loans(member_id)
        if active >= member.max_loans:
            return f"ERROR: Member already has {active} active loans (max {member.max_loans})."

        copy = self.repo.find_available_copy(isbn)
        if not copy:
            return "ERROR: No available copies."

        loan_id = f"LN-{len(self.repo.loans)+1}"
        today = date.today().isoformat()
        due = (date.today() + timedelta(days=14)).isoformat()

        loan = Loan(
            loan_id=loan_id,
            copy_id=copy.copy_id,
            member_id=member_id,
            issue_date=today,
            due_date=due
        )
        self.repo.create_loan(loan)
        self.repo.set_copy_status(copy.copy_id, "Issued")
        return f"SUCCESS: Issued copy {copy.copy_id}, loan {loan_id}, due {due}"

    def return_book(self, copy_id: str, member_id: str) -> str:
        loan = self.repo.find_active_loan(copy_id, member_id)
        if not loan:
            return "ERROR: No active loan found for that copy+member."

        today = date.today().isoformat()
        loan.mark_returned(today)
        self.repo.set_copy_status(copy_id, "Available")

        # Optional: basic overdue check (no money math)
        if today > loan.due_date:
            return f"SUCCESS: Returned (OVERDUE). Due was {loan.due_date}, returned {today}"

        return "SUCCESS: Returned on time."

main.py (demo run)

Python
from models import Book, BookCopy, Member
from repository import Repository
from service import LibraryService

def seed(repo: Repository) -> None:
    repo.add_book(Book(isbn="111", title="OOP Basics", author="A. Author"))
    repo.add_copy(BookCopy(copy_id="C1", isbn="111"))
    repo.add_copy(BookCopy(copy_id="C2", isbn="111"))
    repo.add_member(Member(member_id="M1", name="Sam", max_loans=2))

def main():
    repo = Repository()
    seed(repo)
    service = LibraryService(repo)

    print(service.issue_book("111", "M1"))
    print(service.issue_book("111", "M1"))
    print(service.issue_book("111", "M1"))  # should fail (max loans or no copies)
    print(service.return_book("C1", "M1"))
    print(service.issue_book("111", "M1"))  # should work again

if __name__ == "__main__":
    main()
